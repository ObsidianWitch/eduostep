    ```sh
    experiments() {
        for seed in {0..9}; do
            ./relocation.py --addresses=100 --seed="$seed" --l="$1" -c
        done
    }
    export -f experiments # make function visible to `parallel`

    seq 0 100 16383 \
        | parallel 'experiments {}' \
        | ./plot.py > q5plot.png
    ```
